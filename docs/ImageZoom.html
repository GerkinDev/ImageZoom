(private) initLoaded()

Check if all is ok and end initialization

This function meant to be called when natural image is fully loaded. Detection of ready status is done by reading the "complete" status of the target image, and its natural dimensions. Constructor call this: 
setTimeout(function() {
    if (self.zoomedImage.complete && original.complete && self.zoomedImage.naturalWidth != 0 && self.zoomedImage.naturalHeight != 0) {
        initLoaded();
    } else {
        intervalTestLoaded = setInterval(function() {
            if (self.zoomedImage.complete && original.complete && self.zoomedImage.naturalWidth != 0 && self.zoomedImage.naturalHeight != 0) {
                clearInterval(intervalTestLoaded);
                initLoaded();
            }
        }, 50);
    }
}, 50);
